# 驗證管線

> 讀取時機：驗證階段。三層驗證的完整 API 與程式碼模式。

---

## 三層驗證架構

```
Layer 1: 結構驗證 → 程式化檢查，快速
Layer 2: 視覺品檢 → Excel COM → PDF → PNG，表哥自己看
Layer 3: 相容性閘門 → Google Sheets 安全性檢查
```

任何一層失敗 → 自動修復 → 重驗 → 修不了才報告使用者

## Layer 1: 結構驗證

### 1.1 基礎結構

```python
from openpyxl import load_workbook

def verify_basic_structure(filepath, expected):
    """expected 是 dict，描述預期的結構。"""
    wb = load_workbook(filepath, data_only=False)
    errors = []

    # Sheet 存在性
    for name in expected.get('sheets', []):
        if name not in wb.sheetnames:
            errors.append(f"缺少 Sheet: {name}")

    # 每個 Sheet 的欄數行數
    for ws in wb.worksheets:
        if ws.max_column < expected.get('min_columns', 0):
            errors.append(f"{ws.title}: 欄數不足 ({ws.max_column})")
        if ws.max_row < expected.get('min_rows', 0):
            errors.append(f"{ws.title}: 行數不足 ({ws.max_row})")

    return errors
```

```python
from copy import copy

def verify_format_sample(filepath, checkpoints):
    """
    checkpoints: [
        {"cell": "A1", "sheet": "Sheet1", "font_name": "Arial", "font_bold": True, ...},
        ...
    ]
    """
    wb = load_workbook(filepath)
    errors = []

    for cp in checkpoints:
        ws = wb[cp['sheet']]
        cell = ws[cp['cell']]

        if 'font_name' in cp and cell.font.name != cp['font_name']:
            errors.append(f"{cp['cell']}: 字型應為 {cp['font_name']}，實為 {cell.font.name}")
        if 'font_bold' in cp and cell.font.bold != cp['font_bold']:
            errors.append(f"{cp['cell']}: bold 應為 {cp['font_bold']}")
        if 'font_size' in cp and cell.font.size != cp['font_size']:
            errors.append(f"{cp['cell']}: 字型大小應為 {cp['font_size']}，實為 {cell.font.size}")
        if 'fill_color' in cp:
            actual = cell.fill.fgColor.rgb if cell.fill.fgColor else None
            if actual and actual != cp['fill_color']:
                errors.append(f"{cp['cell']}: 填充色應為 {cp['fill_color']}，實為 {actual}")
        if 'number_format' in cp and cell.number_format != cp['number_format']:
            errors.append(f"{cp['cell']}: 數值格式應為 {cp['number_format']}，實為 {cell.number_format}")
        if 'alignment_h' in cp:
            actual = cell.alignment.horizontal
            if actual != cp['alignment_h']:
                errors.append(f"{cp['cell']}: 水平對齊應為 {cp['alignment_h']}，實為 {actual}")

    return errors
```

### 1.3 公式驗算

```python
def verify_formulas(filepath, formula_checks):
    """
    formula_checks: [
        {"cell": "D10", "sheet": "Sheet1", "expected_formula": "=B10*C10"},
        {"cell": "D20", "sheet": "Sheet1", "expected_value_approx": 1234.56, "tolerance": 0.01},
    ]
    """
    wb = load_workbook(filepath, data_only=False)
    wb_values = load_workbook(filepath, data_only=True)
    errors = []

    for fc in formula_checks:
        ws = wb[fc['sheet']]
        cell = ws[fc['cell']]

        if 'expected_formula' in fc:
            if cell.value != fc['expected_formula']:
                errors.append(f"{fc['cell']}: 公式應為 {fc['expected_formula']}，實為 {cell.value}")

        if 'expected_value_approx' in fc:
            ws_val = wb_values[fc['sheet']]
            actual = ws_val[fc['cell']].value
            if actual is None:
                errors.append(f"{fc['cell']}: 公式結果為 None（可能尚未計算）")
            elif abs(actual - fc['expected_value_approx']) > fc.get('tolerance', 0.01):
                errors.append(f"{fc['cell']}: 值應為 ~{fc['expected_value_approx']}，實為 {actual}")

    return errors
```

### 1.4 資料驗證完整性

```python
def verify_data_validations(filepath, expected_dvs):
    """
    expected_dvs: [
        {"sheet": "Sheet1", "range": "B5:B20", "type": "list", "formula1": '"A,B,C"'},
    ]
    """
    wb = load_workbook(filepath)
    errors = []

    for edv in expected_dvs:
        ws = wb[edv['sheet']]
        found = False
        for dv in ws.data_validations.dataValidation:
            if str(dv.sqref) == edv['range'] and dv.type == edv['type']:
                found = True
                if 'formula1' in edv and dv.formula1 != edv['formula1']:
                    errors.append(f"{edv['range']}: DV formula1 不匹配")
                break
        if not found:
            errors.append(f"{edv['sheet']} {edv['range']}: 缺少資料驗證規則")

    return errors
```

```python
def verify_merged_borders(filepath):
    """檢查所有合併區域的外框是否完整。"""
    wb = load_workbook(filepath)
    errors = []

    for ws in wb.worksheets:
        for merge_range in ws.merged_cells.ranges:
            min_col, min_row, max_col, max_row = merge_range.bounds

            # 檢查上邊
            for col in range(min_col, max_col + 1):
                cell = ws.cell(row=min_row, column=col)
                if not cell.border.top or not cell.border.top.style:
                    errors.append(f"{ws.title} 合併區域 {merge_range}: 上邊框缺失 (col {col})")

            # 檢查下邊
            for col in range(min_col, max_col + 1):
                cell = ws.cell(row=max_row, column=col)
                if not cell.border.bottom or not cell.border.bottom.style:
                    errors.append(f"{ws.title} 合併區域 {merge_range}: 下邊框缺失 (col {col})")

            # 檢查左邊
            for row in range(min_row, max_row + 1):
                cell = ws.cell(row=row, column=min_col)
                if not cell.border.left or not cell.border.left.style:
                    errors.append(f"{ws.title} 合併區域 {merge_range}: 左邊框缺失 (row {row})")

            # 檢查右邊
            for row in range(min_row, max_row + 1):
                cell = ws.cell(row=row, column=max_col)
                if not cell.border.right or not cell.border.right.style:
                    errors.append(f"{ws.title} 合併區域 {merge_range}: 右邊框缺失 (row {row})")

    return errors
```

```python
def verify_other_structural(filepath, expected):
    wb = load_workbook(filepath)
    errors = []

    for ws in wb.worksheets:
        # 凍結窗格
        if 'freeze_panes' in expected:
            exp = expected['freeze_panes'].get(ws.title)
            if ws.freeze_panes != exp:
                errors.append(f"{ws.title}: 凍結窗格應為 {exp}，實為 {ws.freeze_panes}")

        # 列印設定
        if 'print_area' in expected:
            exp = expected['print_area'].get(ws.title)
            if ws.print_area != exp:
                errors.append(f"{ws.title}: 列印區域應為 {exp}，實為 {ws.print_area}")

        # 命名範圍
        if 'defined_names' in expected:
            for name in expected['defined_names']:
                if name not in [dn.name for dn in wb.defined_names.definedName]:
                    errors.append(f"缺少命名範圍: {name}")

        # 隱藏行列
        if 'hidden_rows' in expected:
            for row_idx in expected['hidden_rows'].get(ws.title, []):
                if row_idx not in ws.row_dimensions or not ws.row_dimensions[row_idx].hidden:
                    errors.append(f"{ws.title}: 第 {row_idx} 行應為隱藏")

    return errors
```

### 管線

```python
import win32com.client
import fitz  # PyMuPDF
import os

PY313 = r"<Python 3.13 路徑>"

def render_xlsx_to_png(xlsx_path, output_dir=None):
    """Excel COM → PDF → PyMuPDF → PNG。回傳 PNG 路徑清單。"""
    xlsx_path = os.path.abspath(xlsx_path)
    if output_dir is None:
        output_dir = os.path.dirname(xlsx_path)

    pdf_path = os.path.join(output_dir, os.path.splitext(os.path.basename(xlsx_path))[0] + '.pdf')

    # Step 1: xlsx → PDF via Excel COM
    excel = win32com.client.Dispatch("Excel.Application")
    excel.Visible = False
    excel.DisplayAlerts = False
    try:
        workbook = excel.Workbooks.Open(xlsx_path)
        workbook.ExportAsFixedFormat(0, pdf_path)  # 0 = xlTypePDF
        workbook.Close(False)
    finally:
        excel.Quit()

    # Step 2: PDF → PNG via PyMuPDF
    png_paths = []
    doc = fitz.open(pdf_path)
    for i, page in enumerate(doc):
        mat = fitz.Matrix(2, 2)  # 2x zoom
        pix = page.get_pixmap(matrix=mat)
        png_path = os.path.join(output_dir, f"preview_page_{i}.png")
        pix.save(png_path)
        png_paths.append(png_path)
    doc.close()

    # 清理 PDF
    os.remove(pdf_path)

    return png_paths
```

### 品檢要點

表哥讀 PNG 時檢查：
1. **文字遮蔽** — 有沒有被截斷的文字
2. **邊框連續性** — 邊框有沒有斷裂
3. **對齊一致性** — 同類資料的對齊方式是否統一
4. **視覺層次** — 標題、欄位名、資料列、合計列是否有明確區分
5. **空白區域** — 有沒有不該空白的地方
6. **欄寬合理性** — 太窄或太寬

```python
def check_gsheets_compatibility(filepath):
    """程式化檢查 Google Sheets 相容性。"""
    wb = load_workbook(filepath)
    warnings = []

    for ws in wb.worksheets:
        for row in ws.iter_rows():
            for cell in row:
                # Theme Color 檢查
                if cell.font.color and cell.font.color.theme is not None:
                    warnings.append(f"{ws.title}!{cell.coordinate}: 字型使用 Theme Color")
                if cell.fill.fgColor and cell.fill.fgColor.theme is not None:
                    if cell.fill.patternType and cell.fill.patternType != 'none':
                        warnings.append(f"{ws.title}!{cell.coordinate}: 填充使用 Theme Color")

                # 字型檢查
                SAFE_FONTS = {'Arial', 'Noto Sans TC', 'Noto Sans SC', 'Noto Sans JP',
                             'Roboto', 'Courier New', None}
                if cell.font.name and cell.font.name not in SAFE_FONTS:
                    warnings.append(f"{ws.title}!{cell.coordinate}: 字型 {cell.font.name} 非 Google Sheets 安全字型")

                # Gradient Fill 檢查
                if hasattr(cell.fill, 'type') and cell.fill.type == 'gradient':
                    warnings.append(f"{ws.title}!{cell.coordinate}: 使用 Gradient Fill")

                # 動態陣列公式檢查
                if cell.value and isinstance(cell.value, str) and cell.value.startswith('='):
                    formula = cell.value.upper()
                    banned = ['FILTER(', 'SORT(', 'UNIQUE(', 'SEQUENCE(', 'LAMBDA(', 'LET(']
                    for fn in banned:
                        if fn in formula:
                            warnings.append(f"{ws.title}!{cell.coordinate}: 使用不相容公式 {fn.rstrip('(')}")

    return warnings
```

## 風險偵測（修改既有檔案前）

```python
def assess_risk(filepath):
    """評估既有 Excel 檔案被 openpyxl roundtrip 的風險。"""
    import zipfile
    risks = {'SAFE': [], 'RISKY': [], 'UNSAFE': []}

    # 副檔名檢查
    if filepath.endswith('.xlsm'):
        risks['UNSAFE'].append("致命：.xlsm 檔案的 VBA 會被 openpyxl 剝離")

    # ZIP 層級檢查
    with zipfile.ZipFile(filepath) as z:
        names = z.namelist()
        if 'xl/vbaProject.bin' in names:
            risks['UNSAFE'].append("致命：含 VBA macro")
        if any('activeX' in n for n in names):
            risks['UNSAFE'].append("致命：含 ActiveX 控件")
        if 'xl/connections.xml' in names:
            risks['UNSAFE'].append("致命：含外部資料連線")
        if any('oleObject' in n.lower() for n in names):
            risks['UNSAFE'].append("致命：含 OLE 物件")
        for name in names:
            if name.startswith('xl/worksheets/sheet'):
                content = z.read(name).decode('utf-8', errors='ignore')
                if 'sparklineGroup' in content:
                    risks['UNSAFE'].append("致命：含 Sparklines")
                    break

    # openpyxl 層級檢查
    from openpyxl import load_workbook
    wb = load_workbook(filepath, read_only=True, data_only=False)

    if wb.defined_names and list(wb.defined_names.definedName):
        risks['RISKY'].append(f"注意：含 {len(list(wb.defined_names.definedName))} 個命名範圍（可能損壞）")

    for ws in wb.worksheets:
        if ws.conditional_formatting:
            risks['RISKY'].append(f"注意：{ws.title} 含條件格式（複雜規則可能損壞）")
        if ws.data_validations and ws.data_validations.dataValidation:
            risks['RISKY'].append(f"注意：{ws.title} 含資料驗證（可能被丟棄）")

    wb.close()

    # 判定
    if risks['UNSAFE']:
        return 'UNSAFE', risks
    elif risks['RISKY']:
        return 'RISKY', risks
    else:
        return 'SAFE', risks
```
